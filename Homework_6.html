library(stringr)
Vector_string <- function(x) {
  l <- length(x)
  if (l == 0) return("")
  if (l == 1) return(x)
  if (l == 2) return(str_c(x, collapse = " and "))
  if (l >= 3){ 
    n <- x[l]
    k <- c("and", n)
    x[l] <- str_c(k, collapse = " ")
